Terminology¶

Encode - The 128 facial encodes that describe the face. This is stored as a numpy array of (128,) Image - A numpy image in RGB format, stored as a numpy array dtype=np.uint8 class_number - A class number that represents each collection of faces. Stored as unsigned int.

Key Data Sets¶

Primary Data¶

A set of lists that contains the encodes, image, and class number for each image.

encodes_list - a list of encodes for each of the images

images_list - a list of images, it's corresponding encode can be found in encodes_list at it's index.

class_numbers_list - a list of class numbers that corresponds to each encode and image at it's index.

Secondary Data¶

A set of lists that describe these secondary data that required processing.

best_encodes_list - a list of best encodes, each index corresponds to its class number, and stores the best encode for that class.

best_encodes_index_list - a list of indexes that points to the index number in the encodes_list where the best encode was found.

best_encodes_image_list - a list of images that corresponds to each of the best_encodes.

Metric Functions¶

evaluate_model_best_accuracy

evaluate_model_best_accuracy(model_path, test_data_path) -> pd.DataFrame, float

    Evalutes model accuracy by taking each image and comparing it to the list of best_encodes for matches.
    An image is correct if it matches it's own class, and incorrect otherwise.
    Returns a dataframe that describes the success for each tests as well as how far it is from correct.
    Returns a float that contains accuracy (corrects / (corrects + incorrects)

evaluate_model_universal_metric_scores

evaluate_model_universal_metric_scores(model_path, test_data_path, threshold=0.6) -> pd.DataFrame

    Evaluates each entry in the datapath, and compares it to every other entry in the datapath, using the
    passed threshold. The desired result is images of the same class matches each other.
    Returns a dataframe containing the following data:
        Number of True Positive | Number of True Negative | Number of False Positive | Number of False Negative
        Accuracy    |      Precision      |      Recall     |    Specificity   |       F1

def build_primary_data(rootpath, encoder):
    images_list = []
    class_numbers_list = []

    # look through each file in each subdirectory, read the image and add both the image and class number to the list
    for class_number, class_path in enumerate(rootpath.glob('*')):
        for fpath in class_path.glob('*'):
            images_list.append(np.asarray(Image.open(fpath)))
            class_numbers_list.append(class_number)

    # hash the images_list to see if we already have a cached version saved
    hasher = hashlib.md5()
    # sample encode to "verify" the hash of the model
    sample_encode = encoder.compute_face_descriptor(images_list[0], 1)

    hasher.update(repr([images_list, class_numbers_list, sample_encode]).encode('utf-8'))
    cache_filename = hasher.hexdigest()

    if Path(cache_filename).exists():
        # if cache file exists, just load it
        with open(cache_filename, 'rb') as f:
            encodes_list = pickle.load(f)
    else:
        # if no cache exists, generate the encoding list and then save it.
        encodes_list = np.array([encoder.compute_face_descriptor(image, 1) for image in images_list])
        # write cache to file
        with open(cache_filename, 'wb+') as f:
            pickle.dump(encodes_list, f)

    return np.array(encodes_list), np.array(images_list, dtype=np.uint8), np.array(class_numbers_list, dtype=np.uint32)

def build_secondary_data(encodes_list, images_list, class_numbers_list):
    best_encodes_list = []
    best_encodes_index_list = []
    best_encodes_image_list = []

    def bucket_encodes_by_class_numbers(encodes_list, class_numbers_list):
        it = iter(class_numbers_list)
        sorted_encodes_list = sorted(encodes_list, key=lambda _: next(it))
        sorted_class_numbers_list = sorted(class_numbers_list)
        
        it = iter(sorted_class_numbers_list)
        return [list(g) for k, g in itertools.groupby(sorted_encodes_list, key=lambda _: next(it))]        
    
    # buckets of grouped by class number, from smallest to largest. The first bucket would be class 0 encodes
    class_encode_buckets = bucket_encodes_by_class_numbers(encodes_list, class_numbers_list)

    # dictionary mapping a repr string representation to the index of the encode
    # will be used to map best encode to its index number
    encodes_dictionary = dict(zip(map(lambda t: repr(t.tolist()), encodes_list), range(len(encodes_list))))
    
    
    for class_encodes in class_encode_buckets:
        # calculate the ideal encode (this is just the average)
        ideal_encode = np.average(np.array(class_encodes), axis=0)

        # find the encode with the smallest distance from the ideal, that is our best encode
        min_idx = np.argmin(np.linalg.norm(class_encodes - ideal_encode, axis=1))
        
        # best encode
        best_encode = class_encodes[min_idx]
        
        # find the index of the best encode to find the image of the best encode
        best_encode_idx = encodes_dictionary[repr(best_encode.tolist())]
        
        # best encode image
        best_image = images_list[best_encode_idx]
        
        # append the results to the list
        best_encodes_list.append(best_encode)
        best_encodes_index_list.append(best_encode_idx)
        best_encodes_image_list.append(best_image)
        
    return best_encodes_list, best_encodes_index_list, best_encodes_image_list

def evaluate_model_best_accuracy(model_path, test_data_path):
    encoder = dlib.face_recognition_model_v1(model_path)
    (
        encodes_list, images_list, class_numbers_list,
        best_encodes_list, best_encodes_index_list, best_encodes_image_list
    ) = build_directory_data(Path(test_data_path), encoder)
    
    clean_data_set = list(map(lambda _: False, encodes_list))

    df = pd.DataFrame(
        {
            # True or False, Correct or Incorrect
            'CORRECT': clean_data_set.copy(), 'INCORRECT': clean_data_set.copy(), 
            # Index of the class
            'EXPECTED_CLASS_INDEX': np.array(clean_data_set.copy(), dtype=np.uint32),
            'RESPONSE_CLASS_INDEX': np.array(clean_data_set.copy(), dtype=np.uint32),
            # How far expected best index is from the top position (position 0)
            'EXPECTED_OFFSET': np.array(clean_data_set.copy(), dtype=np.uint32),
            # Index of the tested image/encode and the response.
            'TEST_IMAGE_INDEX': np.array(clean_data_set.copy(), dtype=np.uint32),
            'RESPONSE_IMAGE_INDEX': np.array(clean_data_set.copy(), dtype=np.uint32),
        }
    )

    # calculate the position of the correct_class_index
    def find_correct_class_position(thresholds, class_index):
        # [(index, threshold)....] sorted by threshold
        sorted_thresholds = sorted(enumerate(thresholds), key=lambda t: t[1])

        for position, zthreshold in enumerate(sorted_thresholds):
            # check if the stored index match class_index
            if zthreshold[0] == class_index:
                return position
        # this should never happen
        return False           
    
    for i, encode in enumerate(encodes_list): 

        thresholds = np.linalg.norm(best_encodes_list - encode, axis=1)
        
        # from the list of best encodes, find the best encode that best matches the encode
        # the idx is the class number
        guess_class_idx = np.argmin(thresholds)
        
        # get the class index, this is the class the correct answer to the best_idx
        correct_class_index = class_numbers_list[i]

        # get the correct class position
        correct_class_position = find_correct_class_position(thresholds, correct_class_index)    
        
        # get the image index of the guess
        guess_image_index = best_encodes_index_list[guess_class_idx]

        # the results match the actual index, TRUE POSITIVE
        if guess_class_idx == correct_class_index:
            df.at[i, 'CORRECT'] = True
        else:
            df.at[i, 'INCORRECT'] = True
            
        df.at[i, 'EXPECTED_CLASS_INDEX'] = correct_class_index
        df.at[i, 'RESPONSE_CLASS_INDEX'] = guess_class_idx
        df.at[i, 'EXPECTED_OFFSET'] = correct_class_position
        
        df.at[i, 'TEST_IMAGE_INDEX'] = i
        df.at[i, 'RESPONSE_IMAGE_INDEX'] = guess_image_index

        corrects = df['CORRECT'].sum()
        incorrects = df['INCORRECT'].sum()
        accuracy = corrects / (corrects + incorrects)        
        
    return df, accuracy

def evaluate_model_universal_metric_scores(model_path, test_data_path, threshold=0.6):
    encoder = dlib.face_recognition_model_v1(model_path)
    (
        encodes_list, images_list, class_numbers_list,
        best_encodes_list, best_encodes_index_list, best_encodes_image_list
    ) = build_directory_data(Path(test_data_path), encoder)
    
    clean_numeric_data = list(map(lambda _: 0, encodes_list))
    clean_list_data = list(map(lambda t: [], encodes_list))

    df = pd.DataFrame(
        {
            # True or False, Correct or Incorrect
            'NUMBER_OF_TRUE_POSITIVES': clean_numeric_data.copy(),
            'NUMBER_OF_TRUE_NEGATIVES': clean_numeric_data.copy(),
            'NUMBER_OF_FALSE_POSITIVES': clean_numeric_data.copy(),
            'NUMBER_OF_FALSE_NEGATIVES': clean_numeric_data.copy(),
            
            # Index of the class
            'TRUE_POSITIVE_INDEXES': copy.deepcopy(clean_list_data),
            'TRUE_NEGATIVE_INDEXES': copy.deepcopy(clean_list_data),
            'FALSE_POSITIVE_INDEXES': copy.deepcopy(clean_list_data),
            'FALSE_NEGATIVE_INDEXES': copy.deepcopy(clean_list_data),
        }
    )
    
    for i, encode in enumerate(encodes_list): 
        # get all thresholds
        thresholds = np.linalg.norm(encodes_list - encode, axis=1)        
        
        # get the class of the current encode
        target_class_number = class_numbers_list[i]
        
        for j, encode_threshold in enumerate(thresholds):
            correct_class_number = class_numbers_list[j]
            
            # they belong to the same class, the encode_threshold should be under the threshold:
            if target_class_number == correct_class_number:
                
                # TRUE POSITIVE CONDITION - SAME CLASS AND UNDER THE THRESHOLD
                if encode_threshold <= threshold:
                    df.at[i, 'NUMBER_OF_TRUE_POSITIVES'] += 1
                    df.at[i, 'TRUE_POSITIVE_INDEXES'].append(j)
                # FALSE NEGATIVE CONDITION - SAME CLASS BUT OVER THE THRESHOLD
                else:
                    df.at[i, 'NUMBER_OF_FALSE_NEGATIVES'] += 1
                    df.at[i, 'FALSE_NEGATIVE_INDEXES'].append(j)
                    
            # they do not belong to the same class, the encode_threshold should be over the threshold
            else:
                # FALSE POSITIVE CONDITION - DIFFERENT CLASS BUT UNDER THE THRESHOLD
                if encode_threshold <= threshold:
                    df.at[i, 'NUMBER_OF_FALSE_POSITIVES'] += 1
                    df.at[i, 'FALSE_POSITIVE_INDEXES'].append(j)
                # TRUE NEGATIVE CONDITION - DIFFERENT CLASS AND OVER THE THRESHOLD
                else:
                    df.at[i, 'NUMBER_OF_TRUE_NEGATIVES'] += 1
                    df.at[i, 'TRUE_NEGATIVE_INDEXES'].append(j)                    

    """
    creating evaluation metrics for each row: accuracy, precision, recalls, specificies, f1 score
    add a totals row to ensure for a summary of the entire system
    """
    data = df.to_numpy()

    # generate the totals column
    totals_row = np.concatenate((np.sum(data[:,0:4], axis=0), [np.NAN, np.NAN, np.NAN, np.NAN]))
    df.loc[len(df.index)] = totals_row
    
    # rename the last row to TOTALS
    df.rename(index={len(df.index) - 1: 'TOTALS'}, inplace=True)    
    
    data = df.to_numpy()
    
    # calculate the metrics
    total_number_of_predictions = np.sum(data[:, 0:3], axis=1)
    
    # number of correct predictions (true positive + true negative) / number of predictions made
    accuracies = (data[:,0] + data[:,1]) / total_number_of_predictions
    
    # number of true positives / (number of true positives + number of false positives)
    precisions = data[:,0] / (data[:,0] + data[:,2])
    
    # number of true positives / (number of true positives + number of false negatives)
    recalls = data[:,0] / (data[:,0] + data[:,3])
    
    # number of true negatives / (number of true negatives + false positives)
    specificities = data[:,1] / (data[:,1] + data[:,2])

    # lookup f1 scores
    f1_scores = 2 / ((1 / precisions) + (1 / recalls))

    # add all data to the dataframe
    df['ACCURACIES'] = accuracies
    df['PRECISIONS'] = precisions
    df['RECALLS'] = recalls
    df['SPECIFICITIES'] = specificities
    df['F1_SCORES'] = f1_scores
    
    return df
